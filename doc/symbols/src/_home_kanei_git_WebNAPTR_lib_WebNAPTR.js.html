 11 /**
 12  * Webnaptr class containing all the methods and variables of library
 13  */
 14 Webnaptr = new Object();
 15    order = 0;
 16    preference = 1;
 17    flags = 0;
 18    enumProvider = 0;
 19    replacement= 0;
 20    uri = 0;
 21    tel = 0;
 22    regexp = 0;
 23    ignorecase = 0;
 24 
 25    error = '';
 26 
 27    /**
 28    * Sets all of the common parametres of NAPTR record
 29    *
 30    * @param o order - 16-bit unsigned integer specifying the order in which
 31    * the NAPTR records will be processed.
 32    * @param p preference - 16-bit unsigned integer that specifies the order
 33    * in which NAPTR records with equal "order" values should be processed.
 34    * @param f flags - one of four so far defined flags, "S", "A", "U", and "P".
 35    * @return true if everything went ok, otherwise false
 36    */
 37    setParams = function(o, p, f) {
 38       return this.setOrder(o) && this.setPreference(p) && this.setFlags(f);
 39    };
 40 
 41    /**
 42    * Sets order value for NAPTR record.
 43    *
 44    * A 16-bit unsigned integer specifying the order in which the NAPTR records
 45    * will be processed.
 46    *
 47    * @param o order numeric value
 48    * @return true if number was an correct input, otherwise false
 49    */
 50    setOrder = function(o) {
 51       if((o = parseInt(o))) {
 52          order = o;
 53          return true;
 54       }
 55       else {
 56          this.error = "setOrder: Wrong input data - not of INT type.";
 57          return false;
 58       }
 59    };
 60 
 61    /**
 62    * Sets preference value for NAPTR record.
 63    *
 64    * A 16-bit unsigned integer that specifies the order in which NAPTR records
 65    * with equal "order" values should be processed.
 66    *
 67    * @param p preference numeric value
 68    * @return true if number was an correct input, otherwise false
 69    */
 70    setPreference = function(p) {
 71       if((p = parseInt(p))) {
 72          order = p;
 73          return true;
 74       }
 75       else {
 76          this.error = "setPreference: Wrong input data - not of INT type.";
 77          return false;
 78       }
 79    };
 80 
 81    /**
 82    * Sets flags for NAPTR record.
 83    *
 84    * At this time only four flags, "S", "A", "U", and "P", are defined.
 85    * "S" means that the next lookup should be for SRV records.
 86    * "A" means that the next lookup should be for either an A, AAAA, or A6
 87    * record.
 88    * "U" means that the next step is not a DNS lookup but that the output of
 89    * the Regexp field is an URI that adheres to the 'absoluteURI'.
 90    *
 91    * @param f defines flag character
 92    * @return true if character was accepted, otherwise returns false
 93    */
 94    setFlags = function(f) {
 95       f = f.toUpperCase();
 96       if(f == 'S' || f == 'U' || f == 'A' || f == 'P') {
 97          this.flags = f;
 98          return true;
 99       }
100       else {
101          this.error = "setFlags: Wrong input data - only 's', 'u', 'a' and 'p' letters are allowed.";
102          return false;
103       }
104    };

108    /**
109     * Set telephone number, which is used for constructing NAPTR record
110     * @param tel input telephone number
111     * @return true on success, false on failure
112     */
113    setTelNum = function(tel) {
114       if(parseInt(tel)) {
115          var result = '';
116          for (var i = tel.length-1; i>=0; i--) {
117             result += tel.charAt(i) + ".";
118          }
119          if(this.enumProvider == this.Providers.E164ARPA)
120             result += "e164.arpa.";
121          else if(this.enumProvider == this.Providers.NRENUM)
122             result += "nrenum.net.";
123          this.tel = result;
124          return true;
125       }
126       else {
127          this.tel = 0;
128          this.error = "setTelNum: Wrong input data - not a correct telephone number.";
129          return false;
130       }
131    };
132 
133    /**
134     * Set URI address, which is used for constructing NAPTR record
135     * @param uri address for replacement
136     * @return true on success, false on failure
137     */
138    setURI = function(uri) {
139       if(uri!=''){
140          this.uri = uri;
141          return true;
142       }
143       else {
144          this.error = "setURI: Wrong input data - empty string.";
145          return false;
146       }
147    };

149    /**
150     * Set Replacement
151     * A <domain-name> which specifies the new value in the
152     * case where the regular expression is a simple replacement
153     * operation.
154     */
155    setReplacement = function(rep) {
156       //!! Regexp for email^([0-9a-zA-Z]([-.\w]*[0-9a-zA-Z])*@([0-9a-zA-Z][-\w]*[0-9a-zA-Z]\.)+[a-zA-Z]{2,9})$
157       //@TODO check by regular expreesion
158       this.replacement = rep;
159    };
160 
161    /**
162     * Set first part of regexp - which strings should be affected
163     * @param regexp regexp containing only first part of regular expression
164     */
165    setRegexp = function(regexp) {
166       if(regexp != '')
167          this.regexp = regexp;
168       else
169          this.regexp = 0;
170    };
171    
172    /**
173     * Generates NAPTR record for desired service
174     * @param service_enum defines which service will be generated
175     * enums are available as part of Webnaptr object
176     * @return on success string containing NAPTR record or false on failure
177     */
178    generate  = function(service_enum) {
179       var record = '';
180 
181       if(!this.tel) {
182           this.error = "generate: telephone number has not been set.";
183           return false;
184       }
185       if(!this.uri) {
186           this.error = "generate: URI has not been set.";
187           return false;
188       }
189 
190       //first row
191       //$ORIGIN 2.4.2.4.5.5.5.1.e164.arpa.
192       record += '$origin ' + this.tel + '\n';
193 
194       //second row
195       //IN NAPTR 100 10 "u" "E2U+sip"  "!^.*$!sip:test@test.net!"
196       record += 'IN NAPTR ' + this.order + ' ' + this.preference + ' ';
197       //                      ^order             ^preference
198       record += '"' + this.flags + '" ';
199       //                     ^flags
200 
201       var service;
202       var regServ;
203       
204       //Add rest of record, which depends on every service
205       switch(service_enum) {
206          case this.Services.H323:
207             service = '"E2U+H323"';
208             regServ = 'h323';
209             break;
210          case this.Services.SIP:
211             service = '"E2U+sip"';
212             regServ = 'sip';
213             break;
214          case this.Services.EMAIL:
215             service = '"E2U+email"';
216             regServ = 'mailto';
217             break;
218          case this.Services.IFAX:
219             service = '"E2U+ifax:mailto"';
220             regServ = 'mailto:';
221             break;
222          default:
223             this.error = "Wrong service id has been entered";
224             return false;
225             break;
226       }
227 
228       // form record
229       record += service + ' "!' + (this.regexp ? this.regexp : '^.*$') + '!' 
230             //  ^service           ^regexp                                     
231             + regServ + ':' + this.uri + '!' + (this.ignorecase? 'i' : '') + '"';
232             //^service used in regexp          ^ignorecase
233       //Check for 'U' flag - if present, skip replacement field
234       if(this.flags == 'U') {
235          record += ' .';
236       }
237       else {
238          if(!this.replacement) {
239             this.error = "generate: Replacement string has not been set."
240             return false;
241          }
242          record += ' ' + this.replacement + ' .';
243       }
244 
245       return record;
246    }
